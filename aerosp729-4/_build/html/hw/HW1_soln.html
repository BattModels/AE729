import numpy as np

import matplotlib.pyplot as plt
import niceplots

plt.style.use(niceplots.get_style())
colors = niceplots.get_colors_list()

def OCV(Q , Q_max=5.080457832540185):
    coeffs = [-0.44934919, -0.07263436,  0.03009002, -0.02263346, -0.24777699, -0.26923143, 3.72029854]
    x = Q / Q_max
    result = 0.0
    for i in range(0, len(coeffs)-1):
        result += (coeffs[i] ) * ((2*x -1)**(i+1) - (2*x*i*(1-x))/(2*x-1)**(1-i))  # Higher-order terms

    result += coeffs[len(coeffs)-1] + (8.3145 * 298.15 / 96485.3321) * np.log((1-x)/x)
    return result

Equivalent Circuit: Internal Resistor Model#

C_rates = [0.1, 0.5, 1.0]

avg_res_list = np.zeros(3)

for c_idx, data in enumerate(zip(currents, capacities, measured_voltages)):
    current, capacity, voltage = data
    
    capacity = capacity[1:-1]
    voltage = voltage[1:-1]

    v_loss = OCV(capacity) - voltage
    
    resistance = v_loss / current
    average_resistance = integrate.simpson(resistance, capacity) / (capacity[-1] - capacity[0])
    print(f"Average resistance value for {current} A case: {average_resistance} Ohms")

    avg_res_list[c_idx] = average_resistance
    
    plt.plot(capacity, v_loss, label=f"{current} A")
    plt.xlabel("Capacity")
    plt.ylabel("Overpotential [V]")
    plt.legend()
